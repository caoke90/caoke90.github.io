第12节：数据编辑器

第12节：数据编辑器

功能：

数据映射和逆映射、异步请求配置、数据校验、响应式更新页面

支持的数据类型：

数字、字符、下拉option、plusInt、Switch、Time、plusInt、plusNumber、Color、Textarea

传入数据

 {
    "card_type": "tabview",
    "title": '标题',
    "tabIndex": 0,
    "mkeyArr": [{
      name:'首页',
      mkey:'home',
    },{
      name:'商城',
      mkey:'shop',
    },{
      name:'个人中心',
      mkey:'person',
    }],
  }

根据配置

 'tabview':{
    "help":"tabview:多容器",
    'getMkey':function (card) {
      return card.mkeyArr[card.tabIndex].mkey;
    },
    "editHelp":{
      "title":"标题",
      "tabIndex":"当前tab",
      "mkeyArr":"数据来源",
      "mkeyArr.name":"子标题",
      "mkeyArr.mkey":"容器mkey",
    },
    "dataMap":{
      "title":"title",
      "tabIndex":"tabIndex",
      "mkeyArr":"mkeyArr",
    },
    //属性类型
    "propsType":{
      "tabIndex":"Number",
      "mkeyArr.mkey":"Selector",
      "mkeyArr.mkeyOptions":[
        {key: "home", value: 'home'},
        {key: "shop", value: 'shop'},
        {key: "person", value: 'person'},
      ],
    },
  },
