export default {
  cardsDic:adminJSON,
  getCardData:function getCardData(name) {
    if(typeof name=="number"){
      name="card"+name;
    }
    for(var i=0; i<adminJSON.length; i++){
      if(name==adminJSON[i].card_type){
        const card=JSON.parse(JSON.stringify(adminJSON[i]));
        card.key=0;
        return card;
      }
    }
  }
}
